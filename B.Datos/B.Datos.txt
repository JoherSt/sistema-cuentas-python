CREATE TABLE album_names(
    id BIGINT PRIMARY KEY AUTO_INCREMENT,

CREATE TABLE songs (
    id  BIGINT PRIMARY KEY AUTO_INCREMENT,
     name VARCHAR (50) NOT NULL,

    gender varchar (20) not null
);
